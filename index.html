Supabase Database Setup
Paste your Supabase credentials below. Find them at supabase.com → Settings → API.
These are stored locally and only used to connect to your own database.
▶ Show required SQL — run this once in Supabase SQL Editor
Security note: These policies grant the anon role full access so the dashboard can read and write data using your public key. The app enforces role-based permissions client-side. Do not expose this dashboard publicly without adding server-side auth. 
-- ── APM Dashboard — Run once in Supabase SQL Editor ──────────

-- 1. TABLES

create table if not exists users (
  id bigserial primary key,
  username text unique not null,
  display_name text,
  role text not null default 'apm',
  hash text not null
);

create table if not exists projects (
  id bigserial primary key,
  name text not null,
  client text,
  number text,
  status text default 'Planning',
  start_date text,
  end_date text,
  notes text,
  contract_type text,
  contract_value numeric,
  address text,
  owner_name text,
  owner_phone text,
  owner_email text,
  assigned_pm text,
  assigned_apm text,
  assigned_super text,
  assigned_foreman text
);

create table if not exists rfis (
  id bigserial primary key,
  num text,
  project text,
  subject text,
  submitted text,
  due text,
  status text default 'Open',
  priority text default 'Low',
  description text,
  spec_section text,
  closed_at timestamptz,
  source_url text
);

create table if not exists submittals (
  id bigserial primary key,
  num text,
  project text,
  description text,
  spec text,
  submitted_date text,
  need_date text,
  status text default 'Pending',
  bic text,
  approved_at timestamptz,
  source_url text
);

create table if not exists rental (
  id bigserial primary key,
  name text not null,
  project text,
  vendor text,
  onsite_date text,
  release_date text,
  status text default 'On Site'
);

create table if not exists activity_log (
  id bigserial primary key,
  entry_id text,        -- client-generated stable ID for cross-device seen tracking
  type text,
  action text,
  label text,
  detail text,
  username text,
  ts text
);

-- 2. ENABLE ROW LEVEL SECURITY

alter table users        enable row level security;
alter table projects     enable row level security;
alter table rfis         enable row level security;
alter table submittals   enable row level security;
alter table rental       enable row level security;
alter table activity_log enable row level security;

-- 3. RLS POLICIES (anon role — matches your publishable/anon key)

-- users: full access (login, user management)
create policy "anon_select_users" on users for select using (true);
create policy "anon_insert_users" on users for insert with check (true);
create policy "anon_update_users" on users for update using (true) with check (true);
create policy "anon_delete_users" on users for delete using (true);

-- data tables: full access
create policy "anon_all_projects"   on projects   for all using (true) with check (true);
create policy "anon_all_rfis"       on rfis        for all using (true) with check (true);
create policy "anon_all_submittals" on submittals  for all using (true) with check (true);
create policy "anon_all_rental"     on rental      for all using (true) with check (true);

-- activity_log: read + insert (clients can read the feed; no update/delete)
create policy "anon_select_activity" on activity_log for select using (true);
create policy "anon_insert_activity" on activity_log for insert with check (true);

-- 4. IF UPGRADING: add missing project columns to existing table
alter table projects add column if not exists contract_type text;
alter table projects add column if not exists contract_value numeric;
alter table projects add column if not exists address text;
alter table projects add column if not exists owner_name text;
alter table projects add column if not exists owner_phone text;
alter table projects add column if not exists owner_email text;
alter table projects add column if not exists assigned_pm text;
alter table projects add column if not exists assigned_apm text;
alter table projects add column if not exists assigned_super text;
alter table projects add column if not exists assigned_foreman text;

-- IF UPGRADING activity_log: add entry_id column for cross-device seen tracking
alter table activity_log add column if not exists entry_id text;

-- IF UPGRADING rfis/submittals: add terminal-state timestamps used by the
-- "Recently Approved" 7-day section on the RFI and Submittal tabs
alter table rfis       add column if not exists closed_at   timestamptz;
alter table submittals add column if not exists approved_at timestamptz;

-- IF UPGRADING rfis/submittals: add external source URL (link back to Procore/ACC/Bluebeam/etc.)
alter table rfis       add column if not exists source_url text;
alter table submittals add column if not exists source_url text;

-- ── Project detail tables ───────────────────────────────────

create table if not exists milestones (
  id             bigserial primary key,
  project_id     bigint references projects(id) on delete cascade,
  name           text not null,
  milestone_date date,
  milestone_end_date date,
  is_tbd         boolean default false,
  type           text default 'custom',
  notes          text,
  created_by     text,
  created_at     timestamptz default now()
);

create table if not exists subcontractors (
  id             bigserial primary key,
  project_id     bigint references projects(id) on delete cascade,
  company_name   text not null,
  trade          text,
  contact_name   text,
  contact_phone  text,
  contact_email  text,
  status         text default 'Active',
  created_by     text,
  created_at     timestamptz default now()
);

alter table milestones     enable row level security;
alter table subcontractors enable row level security;
create policy "anon_all_milestones"     on milestones     for all using (true) with check (true);
create policy "anon_all_subcontractors" on subcontractors for all using (true) with check (true);

-- ── Material Log tables ─────────────────────────────────────

create table if not exists vendors (
  id             bigserial primary key,
  name           text not null,
  contact_name   text,
  email          text,
  phone          text,
  category       text,
  lead_time_days integer default 7,
  notes          text
);

create table if not exists materials (
  id                bigserial primary key,
  project           text,
  name              text not null,
  description       text,
  quantity          numeric,
  unit              text default 'each',
  status            text default 'Needed',
  vendor_id         bigint references vendors(id) on delete set null,
  vendor_name       text,
  date_needed       text,
  date_ordered      text,
  expected_delivery text,
  actual_delivery   text,
  po_number         text,
  unit_price        numeric,
  total_price       numeric,
  lead_time_days    integer,
  notes             text,
  ordered_by        text
);

alter table vendors   enable row level security;
alter table materials enable row level security;
create policy "anon_all_vendors"   on vendors   for all using (true) with check (true);
create policy "anon_all_materials" on materials for all using (true) with check (true);

-- ── Procurement table ──────────────────────────────────────

create table if not exists procurement (
  id                    bigserial primary key,
  project               text not null,
  name                  text not null,
  status                text default 'Pending',
  order_date            date,
  shipping_date         date,
  requested_onsite_date date,
  vendor                text,
  vendor_contact        text,
  notes                 text,
  created_by            text,
  created_at            timestamptz default now()
);

alter table procurement enable row level security;
create policy "anon_all_procurement" on procurement for all using (true) with check (true);
